# Phansora API — consolidated dependencies (spokenverse + chrono_origin + dossier_nova)
#
# Torch baseline: aligned to the CosyVoice2 TTS engine's vLLM backend (torch 2.7.0,
# CUDA 12.6). vLLM 0.9.0 HARD-PINS torch==2.7.0 (and pulls xformers/torchvision that
# require the same), so torch is 2.7.0 here — NOT 2.8. torchvision 0.22.0 / torchaudio
# 2.7.0 are the matching releases. Prod runs on an RTX A4000 (16 GB, Ampere) — plenty
# for CosyVoice2 (~3 GB fp16). `+cu126` runs on any recent driver via CUDA minor-version
# compatibility. For a CPU-only host, swap `whl/cu126` + `+cu126` for `whl/cpu` + `+cpu`
# and set COSYVOICE2_USE_VLLM=0 / _USE_TRT=0 (both are CUDA-only).
#
# System packages required: ffmpeg, tesseract-ocr
#
# TTS engine: CosyVoice2 is a git checkout (NOT a published pip package), run in-process.
# In prod: run `make install` (installs the torch/vllm/transformers pins below), then
# clone FunAudioLLM/CosyVoice (+ its Matcha-TTS submodule), install its requirements INTO
# this venv with torch/torchaudio/pydantic STRIPPED (a constraints/strip is required — its
# requirements pin torch==2.3.1, which would fight vLLM; see the Makefile), download the
# CosyVoice2-0.5B checkpoints, and set COSYVOICE2_REPO. The API boots without it; only TTS
# calls require it. ⚠ vLLM 0.9.0 needs transformers==4.51.3 and pydantic>=2.9 — both
# pinned below so the CosyVoice requirements install can't downgrade them.
# ─────────────────────────────────────────────────────────────────────────────

# Torch wheels are pinned as DIRECT CloudFront URLs (download.pytorch.org) rather than
# via `--extra-index-url .../whl/cu126`. The index now hands out download-r2.pytorch.org
# (Cloudflare R2) links, and the prod network cannot complete a TLS handshake to
# Cloudflare (SSLV3_ALERT_HANDSHAKE_FAILURE / alert 40, IPv4 + IPv6). The
# download.pytorch.org path serves the same wheels from Amazon CloudFront, which is
# reachable. These URLs are pinned to cp310 / manylinux_2_28_x86_64 — the prod image
# (torch >= 2.7 uses the manylinux_2_28 tag, NOT the old linux_x86_64). For a CPU-only
# host, swap `whl/cu126` + `+cu126` for `whl/cpu` + `+cpu`.
torch @ https://download.pytorch.org/whl/cu126/torch-2.7.0%2Bcu126-cp310-cp310-manylinux_2_28_x86_64.whl
torchvision @ https://download.pytorch.org/whl/cu126/torchvision-0.22.0%2Bcu126-cp310-cp310-manylinux_2_28_x86_64.whl
torchaudio @ https://download.pytorch.org/whl/cu126/torchaudio-2.7.0%2Bcu126-cp310-cp310-manylinux_2_28_x86_64.whl

# TTS acceleration — vLLM runs CosyVoice2's autoregressive LLM (CUDA graphs + paged
# attention remove the per-token CPU sync that otherwise starves the GPU; this is the
# dominant synthesis cost). vLLM 0.9.0 is the version validated against CosyVoice2's
# legacy engine (0.10.x untested, <0.9.0 unsupported) and keeps transformers at 4.51.3.
# It hard-pins torch==2.7.0 and needs pydantic>=2.9 (both pinned in this file). CUDA-only;
# a CPU host must set COSYVOICE2_USE_VLLM=0. Its CUDA ops ship precompiled in the wheel —
# no nvcc/CUDA-toolkit needed at runtime (unlike the old DeepSpeed path).
vllm==0.9.0

# Shared numeric / runtime pins. filelock/networkx upper caps were for the old torch-2.8
# stack; the CosyVoice2 vLLM stack needs newer: vLLM 0.9.0 requires filelock>=3.16.1, and
# CosyVoice's own requirements pin networkx==3.1 (so allow 3.x or `make install-tts` fails).
numpy<2
typing-extensions>=4.14.1
filelock>=3.16.1
networkx>=3.1,<4
